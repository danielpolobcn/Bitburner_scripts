// TARGET SERVER MUST BE ENTERED AS AN ARGUMENT, i.e. "run basic-hack.script n00dles"
var target = args[0];

// Hack will be attempted if current security in the server is less than 10 points above its maximum level.
var MaxSecurity = getServerMinSecurityLevel(target) + 10;
// Hack will be attempted if current money in the server is 70% of maximum money.
var MinMoney = getServerMaxMoney(target) * 0.7;

while (true) {
    if (getServerSecurityLevel(target) > MaxSecurity) {
        weaken(target);
    } else if (getServerMoneyAvailable(target) < MinMoney) {
        grow(target);
    } else {
        hack(target);
    }
}
